    4. 
                
  4.     	String custCode = "xxxxxx";
    5. 
                
  5.         // [必填] 账号密码
    6. 
                
  6.     	String password = "xxxxxxxxx";
    7. 
                
  7.         // [必填] https://ip:port
    8. 
                
  8.     	String serviceBaseUrl = "https://ip:port";
    9. 
                
  9.     	/**
    10. 
                
  10.     	 * 1.通过SmsReq对象传参
    11. 
                
  11.     	 */
    12. 
                
  12.     	//变量参数
    13. 
                
  13.     	List<VariantParams> variantParamsList = new ArrayList<VariantParams>();
    14. 
                
  14.     	//对应内容中的${var1}和${var2}
    15. 
                
  15.     	String[] vars = {"福州","21"}; 
    16. 
                
  16.     	//对应内容中的${var1}和${var2}
    17. 
                
  17.     	String[] vars1 = {"哈尔滨","-21"}; 
    18. 
                
  18.     	//对应内容中的${var1}和${var2}
    19. 
                
  19.     	String[] vars2 = {"海南","30"}; 
    20. 
                
  20. 
                
  21.     	VariantParams variantParams = new VariantParams();
    22. 
                
  22.     	variantParams.setMobile("13489080110");
    23. 
                
  23.     	variantParams.setVars(vars);
    24. 
                
  24.     	variantParamsList.add(variantParams);
    25. 
                
  25. 
                
  26.     	VariantParams variantParams1 = new VariantParams();
    27. 
                
  27.     	variantParams1.setMobile("13489080111");
    28. 
                
  28.     	variantParams1.setVars(vars1);
    29. 
                
  29.     	variantParamsList.add(variantParams1);
    30. 
                
  30. 
                
  31.     	VariantParams variantParams2 = new VariantParams();
    32. 
                
  32.     	variantParams2.setMobile("13489080112");
    33. 
                
  33.     	variantParams2.setVars(vars2);
    34. 
                
  34.     	variantParamsList.add(variantParams2);
    35. 
                
  35.     	//变量短信请求
                
  36.     	VariantSmsReq variantSmsReq = new VariantSmsReq();
    37. 
                
  37.     	// [选填] 业务标识，由贵司自定义32为数字透传至我司
    38. 
                
  38.     	variantSmsReq.setUid("1123344567")
    39. 
                
  39.     	// [必填] 用户账号
    40. 
                
  40.     	variantSmsReq.setCust_code(custCode)
    41. 
                
  41.     	//[必填] 短信模板。其中的变量用“${vari}”来替代,i代表变量的序号（从0开始增长，每次增加1）。例如：“${var0}用户您好，今天${var1}的天气，晴，温度${var2}度，事宜外出。”，该短信中具有两个变量参数。编码为UTF-8格式。
    42. 
                
  42.     	variantSmsReq.setContent("${var1}天气${var2}度")
    43. 
                
  43.     	// [必填] 同时发送给多个号码时,号码之间用英文半角逗号分隔(,),，其中变量短信每一组为json格式，如: [{"mobile"："手机号码","var":["福州","30"]},{"mobile"："手机号码","var":["厦门","32"]}]	每组变量中第一个变量固定为目标手机号码，对应短信模板中的参数，var为变量个数要与内容中的(其中i变量系好，从1开始)个数匹配，以此类推。（请注意：变量中不要包含有逗号和竖线，否则发送的格式解析会有问题）
    44. 
                
  44.     	variantSmsReq.setParams(variantParamsList)
    45. 
                
  45.     	// [选填] 状态报告需求与否，是 yes 否 no 默认yes
    46. 
                
  46.     	variantSmsReq.setNeed_report("yes")
    47. 
                
  47.     	// [选填] 长号码
    48. 
                
  48.     	variantSmsReq.setSp_code("")
    49. 
                
  49.     	// [选填] 信息格式，0：ASCII串；3：短信写卡操作；4：二进制信息；8：UCS2编码；默认8
    50. 
                
  50.     	variantSmsReq.setMsgFmt("8")
    51. 
                
  51. 
                
  52.     	SmsClient smsClient = new SmsClient();
    53. 
                
  53.     	ResultMsg resultMsg = smsClient.sendVariantSms(variantSmsReq, password, serviceBaseUrl);
    54. 
                
  54.     	if (resultMsg.isSuccess()) {
    55. 
                
  55.     		System.out.println(resultMsg.getData());
    56. 
                
  56.     	} else {
    57. 
                
  57.     		System.out.println(resultMsg.getCode());
    58. 
                
  58.     		System.out.println(resultMsg.getMsg());
    59. 
                
  59.     	}
